git init:
    init current folder as local git respository.

git add README.txt
    add file 'README.txt' to git respository.

git add .
    add all files that are not included in git respository to git respository.

git commit -m "comment"
    commit all files to respository with comment.

git remote add origin git@github.com:xxx/xxx.git
    link with a remote github respository.

git push -u origin master
    push local content to remote respository.
